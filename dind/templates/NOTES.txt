1. Get the application URL by running these commands:

{{- if contains "LoadBalancer" .Values.service.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch the status of by running 'kubectl get svc -w {{ include "dind.fullname" . }}'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "dind.fullname" . }} -o jsonpath='{.status.loadBalancer.ingress[0].ip}')
  echo http://$SERVICE_IP:{{ .Values.service.port }}

{{- if contains "ClusterIP" .Values.service.type }}
  export SERVICE_URL=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "dind.fullname" . }} -o jsonpath='{.metadata.name}')
  echo "You have to configure your consumer with : http://${SERVICE_URL}:{{ .Values.service.port }}"
{{- end }}
